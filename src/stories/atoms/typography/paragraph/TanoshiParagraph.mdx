# Paragraph

Use a paragraph atom component to add text content to your page. This will also be used for the SEO.

## Basic usage

```html

<script lang="ts">
    import { TanoshiParagraph, TanoshiParagraphModel } from 'tanoshi';

    const link: TanoshiParagraphModel = new TanoshiParagraphModel('content');

</script>

<TanoshiParagraph TanoshiParagraphModel={link} />

```

<Canvas of={TanoshiParagraphStories.Dark} />

## How to customize

### Themes


```html

<script lang="ts">
    import { TanoshiParagraph, TanoshiParagraphModel } from 'tanoshi';
    import { THEMES } from 'tanoshi';

    const link: TanoshiParagraphModel = new TanoshiParagraphModel('content').setTheme(THEMES.Danger);

</script>

<TanoshiParagraph tanoshiParagraphModel={link} />

```

<Canvas of={TanoshiParagraphStories.Danger} />

### Size
 

```html

<script lang="ts">
    import { TanoshiParagraph, TanoshiParagraphModel } from 'tanoshi';
    import { SIZES } from 'tanoshi';

    const link: TanoshiParagraphModel = new TanoshiParagraphModel('content').setSize(SIZES.Sm);

</script>

<TanoshiParagraph tanoshiParagraphModel={link} />

```

<Canvas of={TanoshiParagraphStories.Small} />
